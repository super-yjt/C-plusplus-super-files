#### Simple Summary<br/>`Doesn't include content in function`

```c++ {.line-numbers}
int main(void)
{
    Addressbooks abs;
    abs.m_Size = 0;
    int select = 0;
    while (true)
    {
        showMenu();
        cin >> select;
        switch (select)
        {
        case 1:
            addPerson(&abs);
            break;
        case 2:
            showPerson(&abs);
            break;
        case 3:
            deletPerson(&abs);
            break;
        case 4:
            findPerson(&abs);
            break;
        case 5:
            modifyPerson(&abs);
            break;
        case 6:
            cleanPerson(&abs);
            break;
        case 0:
            cout << "Welcome to use it next time\n";
            return 0;
            system("pause");
            break;
        default:
            cout << "input error\n";
            break;
        }
    }
}
```
