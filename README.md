#### Simple Summary`Does not include content in function`:
```
int main(void)
{
    Addressbooks abs;
    abs.m_Size = 0;
    int select = 0;
    while (true)
    {
        showMenu();
        cin >> select;
        switch (select)
        {
        case 1:
            addPerson(&abs);
            break;
        case 2:
            showPerson(&abs);
            break;
        case 3:
            deletPerson(&abs);
            break;
        case 4:
            findPerson(&abs);
            break;
        case 5:
            modifyPerson(&abs);
            break;
        case 6:
            cleanPerson(&abs);
            break;
        case 0:
            cout << "欢迎下次使用\n";
            return 0;
            system("pause");
            break;
        default:
            cout << "输入错误\n";
            break;
        }
    }
}
```
